92 public override void ProcessPointerEvent(PointerEvent evt) {

118 // Whenever we change the number of grab points, we save the
119 // current transform data
120 private void BeginTransform() {
121 // End the transform on any existing transformer before we
122 // begin the new one
123 EndTransform();
124
125 int useGrabPoints = GrabPoints.Count;
126 if (_maxGrabPoints != -1) {
127 useGrabPoints = Mathf.Min(useGrabPoints, _maxGrabPoints);
128 }
129
130 switch (useGrabPoints) {
131 case 1:
132 _activeTransformer = OneGrabTransformer;
133 break;
134
135 case 2:
136 _activeTransformer = TwoGrabTransformer;
137 break;
138
139 default:
140 _activeTransformer = null;
141 break;
142 }
143
144 if (_activeTransformer == null) {
145 return;
146 }
147
148 _activeTransformer.BeginTransform();
149 }
150
151 private void UpdateTransform() {
152 if (_activeTransformer == null) {
153 return;
154 }
155
156 _activeTransformer.UpdateTransform();
157 }
158
159 private void EndTransform() {
160 if (_activeTransformer == null) {
161 return;
162 }
163
164 _activeTransformer.EndTransform();
165 _activeTransformer = null;
166 }
167
168 protected override void OnDisable() {
169 if (_started) {
170 EndTransform();
171 }
172
173 base.OnDisable();
174 }
