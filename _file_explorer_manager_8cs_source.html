123 public async Task<Tuple<bool, GameObject>> Import(string url, Pose pose) {
124 string extension = url.Split('.')[^1];
125 if (extension is "gltf" or "glb") {
126 GameObject prefab = Instantiate(
127 _importPrefabs.model3dPrefab,
128 pose.position,
129 pose.rotation,
130 _elementContainerGO.transform);
131 return new Tuple<bool, GameObject>(await prefab.GetComponent<Model3D>().Init(url), prefab);
132 }
133
134 if (extension is "png" or "jpg" or "jpeg") {
135 GameObject prefab = Instantiate(_importPrefabs.imagePrefab, _elementContainerGO.transform);
136 return new Tuple<bool, GameObject>(await prefab.GetComponent<Image>().Init(url), prefab);
137 }
138
139 return new Tuple<bool, GameObject>(false, null);
140 }
